242  void DecodeResourceLimits(TBuiltInResource* resources, char* config)
243  {
244  static const char* delims = " \t\n\r";
245 
246  size_t pos = 0;
247  std::string configStr(config);
248 
249  while ((pos = configStr.find_first_not_of(delims, pos)) != std::string::npos) {
250  const size_t token_s = pos;
251  const size_t token_e = configStr.find_first_of(delims, token_s);
252  const size_t value_s = configStr.find_first_not_of(delims, token_e);
253  const size_t value_e = configStr.find_first_of(delims, value_s);
254  pos = value_e;
255 
256  // Faster to use compare(), but prefering readability.
257  const std::string tokenStr = configStr.substr(token_s, token_e - token_s);
258  const std::string valueStr = configStr.substr(value_s, value_e - value_s);
259 
260  if (value_s == std::string::npos || !(valueStr[0] == '-' || isdigit(valueStr[0]))) {
261  printf("Error: '%s' bad .conf file. Each name must be followed by one number.\n",
262  valueStr.c_str());
263  return;
264  }
265 
266  const int value = std::atoi(valueStr.c_str());
